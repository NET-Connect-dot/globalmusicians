  <script>
    document.getElementById('register-form').addEventListener('submit', function(e) {
      e.preventDefault();
      const formData = new FormData(this);
      const musician = {
        name: formData.get('name'),
        email: formData.get('email'),
        password: formData.get('password'), // Hash in production
        instrument: formData.get('instrument'),
        genre: formData.get('genre'),
        city: formData.get('city'),
        lat: parseFloat(formData.get('lat')) || null,
        lng: parseFloat(formData.get('lng')) || null,
        bio: formData.get('bio'),
        social: formData.get('social'),
        collaboration: formData.get('collaboration') === 'on',
        session_ideas: formData.get('session_ideas')
      };
      let musicians = JSON.parse(localStorage.getItem('musicians') || '[]');
      musicians.push(musician);
      localStorage.setItem('musicians', JSON.stringify(musicians));
      alert('Registration successful! You are now on the map.');
      window.location.href = 'index.html';
    });
  </script>
